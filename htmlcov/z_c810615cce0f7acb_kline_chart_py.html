23def compress_image(filepath: str, max_size: int = MAX_FILE_SIZE) -> str: 

24 """ 

25 使用PIL二次压缩图片 

26  

27 Args: 

28 filepath: 图片文件路径 

29 max_size: 最大文件大小（字节） 

30  

31 Returns: 

32 str: 压缩后的文件路径 

33 """ 

34 try: 

35 from PIL import Image 

36 

37 # 获取当前文件大小 

38 current_size = os.path.getsize(filepath) 

39 if current_size <= max_size: 

40 return filepath 

41 

42 # 打开图片 

43 img = Image.open(filepath) 

44 

45 # 转换为RGB（去除透明通道） 

46 if img.mode in ('RGBA', 'LA', 'P'): 

47 img = img.convert('RGB') 

48 

49 # 首先尝试PNG优化 

50 img.save(filepath, 'PNG', optimize=True) 

51 

52 # 如果仍然超过限制，使用JPEG压缩 

53 if os.path.getsize(filepath) > max_size: 

54 # 尝试不同的质量级别 (更激进的压缩) 

55 for quality in [70, 60, 50, 40, 30]: 

56 img.save(filepath, 'JPEG', quality=quality, optimize=True) 

57 if os.path.getsize(filepath) <= max_size: 

58 break 

59 

60 final_size = os.path.getsize(filepath) 

61 print(f" 图片压缩: {current_size/1024:.1f}KB -> {final_size/1024:.1f}KB") 

62 

63 except ImportError: 

64 # PIL不可用，跳过压缩 

65 pass 

66 except Exception as e: 

67 print(f" 图片压缩失败: {e}") 

68 

69 return filepath

72def generate_kline_chart( 

73 stock_code: str, 

74 stock_name: str, 

75 df: pd.DataFrame, 

76 category: str, 

77 params: dict, 

78 key_candle_dates: list, 

79 output_dir: str = '/tmp/kline_charts', 

80 show_text: bool = False, 

81 show_legend: bool = True 

82) -> str: 

83 """ 

84 生成K线图 

85  

86 Args: 

87 stock_code: 股票代码 

88 stock_name: 股票名称 

89 df: M天历史数据DataFrame (需包含: date, open, close, high, low, volume, short_term_trend, bull_bear_line) 

90 category: 分类（bowl_center, near_duokong, near_short_trend） 

91 params: 策略参数字典 

92 key_candle_dates: 关键K线日期列表 

93 output_dir: 输出目录 

94 show_text: 是否显示文字（默认False，不显示以节省空间） 

95 show_legend: 是否显示图例（默认True） 

96  

97 Returns: 

98 str: 生成的图片文件路径 

99 """ 

100 # 创建输出目录 

101 Path(output_dir).mkdir(parents=True, exist_ok=True) 

102 

103 # 分类名称映射 

104 category_names = { 

105 'bowl_center': 'Bowl Center', 

106 'near_duokong': 'Near DuoKong', 

107 'near_short_trend': 'Near Short Trend' 

108 } 

109 category_name = category_names.get(category, category) 

110 

111 # 准备数据 - 确保按日期正序排列（从早到晚） 

112 df = df.copy() 

113 if pd.api.types.is_datetime64_any_dtype(df['date']): 

114 df['date'] = pd.to_datetime(df['date']) 

115 else: 

116 df['date'] = pd.to_datetime(df['date']) 

117 

118 # 按日期排序（正序）- 注意：需要在排序后重新计算趋势线 

119 df = df.sort_values('date').reset_index(drop=True) 

120 

121 # 重新计算趋势线（需要足够的历史数据）- 优化：如果已有则跳过 

122 # MA114需要至少114天数据才能正确计算 

123 if 'close' in df.columns and len(df) >= 114: 

124 # 检查是否已有趋势线数据（策略已计算过） 

125 if 'short_term_trend' not in df.columns or 'bull_bear_line' not in df.columns: 

126 # 短期趋势线 = EMA(EMA(CLOSE,10),10) 

127 ema10 = df['close'].ewm(span=10, adjust=False, min_periods=1).mean() 

128 df['short_term_trend'] = ema10.ewm(span=10, adjust=False, min_periods=1).mean() 

129 

130 # 多空线 = (MA14 + MA28 + MA57 + MA114) / 4 

131 ma14 = df['close'].rolling(window=14, min_periods=1).mean() 

132 ma28 = df['close'].rolling(window=28, min_periods=1).mean() 

133 ma57 = df['close'].rolling(window=57, min_periods=1).mean() 

134 ma114 = df['close'].rolling(window=114, min_periods=1).mean() 

135 df['bull_bear_line'] = (ma14 + ma28 + ma57 + ma114) / 4 

136 

137 # 只显示最近M天（默认20天），但用全部数据计算了双线 

138 M = params.get('M', 20) 

139 if len(df) > M: 

140 df = df.tail(M).reset_index(drop=True) 

141 else: 

142 print(f"警告: 数据不足{len(df)}天，需要114天才能正确计算多空线") 

143 

144 # 设置图表样式 (120dpi，清晰显示) 

145 if show_text: 

146 # 显示文字版本：保留顶部参数区域 

147 fig = plt.figure(figsize=(10, 8), dpi=120) 

148 gs = fig.add_gridspec(3, 1, height_ratios=[1.2, 5, 2], hspace=0.05) 

149 

150 # ========== 顶部：策略参数 ========== 

151 ax_params = fig.add_subplot(gs[0]) 

152 ax_params.axis('off') 

153 

154 # 格式化参数显示 

155 cap = params.get('CAP', 4000000000) 

156 cap_display = f"{cap/1e8:.0f}B" if cap >= 1e8 else f"{cap/1e4:.0f}W" 

157 

158 param_text = f"Strategy Params | N={params.get('N', 4)} M={params.get('M', 15)} J_VAL={params.get('J_VAL', 30)} CAP={cap_display}" 

159 if 'duokong_pct' in params: 

160 param_text += f" duokong_pct={params.get('duokong_pct')}%" 

161 if 'short_pct' in params: 

162 param_text += f" short_pct={params.get('short_pct')}%" 

163 

164 ax_params.text(0.5, 0.7, param_text, ha='center', va='center', 

165 fontsize=9, fontweight='bold', transform=ax_params.transAxes) 

166 

167 # 股票标题 

168 title_text = f"{stock_code} {stock_name} - {category_name}" 

169 ax_params.text(0.5, 0.25, title_text, ha='center', va='center', 

170 fontsize=11, fontweight='bold', color='#2c3e50', transform=ax_params.transAxes) 

171 

172 # ========== 中部：K线图 ========== 

173 ax_kline = fig.add_subplot(gs[1]) 

174 else: 

175 # 无文字版本：120dpi，清晰显示 

176 fig = plt.figure(figsize=(10, 7), dpi=120) 

177 gs = fig.add_gridspec(2, 1, height_ratios=[3, 1], hspace=0.08) 

178 

179 # ========== 上部：K线图（占更大比例） ========== 

180 ax_kline = fig.add_subplot(gs[0]) 

181 

182 ax_kline.set_xlim(-0.5, len(df) - 0.5) 

183 

184 # 计算价格范围（包含上下影线） 

185 price_cols = ['high', 'low'] 

186 if 'short_term_trend' in df.columns: 

187 price_cols.append('short_term_trend') 

188 if 'bull_bear_line' in df.columns: 

189 price_cols.append('bull_bear_line') 

190 

191 price_max = df[price_cols].max().max() 

192 price_min = df[price_cols].min().min() 

193 price_range = price_max - price_min 

194 ax_kline.set_ylim(price_min - price_range * 0.1, price_max + price_range * 0.15) 

195 

196 # 绘制K线 

197 width = 0.6 

198 for i, row in df.iterrows(): 

199 is_up = row['close'] >= row['open'] 

200 color = '#e74c3c' if is_up else '#27ae60' # 涨红跌绿 

201 

202 # 实体 

203 height = abs(row['close'] - row['open']) 

204 if height == 0: 

205 height = price_range * 0.005 # 最小显示高度（十字星） 

206 bottom = min(row['close'], row['open']) 

207 rect = Rectangle((i - width/2, bottom), width, height, 

208 facecolor=color, edgecolor=color, linewidth=0.5) 

209 ax_kline.add_patch(rect) 

210 

211 # 影线 

212 ax_kline.plot([i, i], [row['low'], row['high']], color=color, linewidth=0.8) 

213 

214 # 关键K线标记（星号） 

215 if row['date'] in key_candle_dates or (isinstance(row['date'], pd.Timestamp) and row['date'].strftime('%Y-%m-%d') in [d.strftime('%Y-%m-%d') if isinstance(d, pd.Timestamp) else d for d in key_candle_dates]): 

216 ax_kline.scatter(i, row['high'] + price_range * 0.03, marker='*', 

217 s=200, color='#f39c12', zorder=5) 

218 

219 # 绘制趋势线（策略里的两根线）- 蓝色(短期)和绿色(多空)实线，加粗 

220 if 'short_term_trend' in df.columns: 

221 ax_kline.plot(range(len(df)), df['short_term_trend'], color='#0066FF', 

222 linewidth=2.5, linestyle='-', label='Short Trend', alpha=0.95) 

223 if 'bull_bear_line' in df.columns: 

224 ax_kline.plot(range(len(df)), df['bull_bear_line'], color='#00AA00', 

225 linewidth=2.5, linestyle='-', label='DuoKong Line', alpha=0.95) 

226 

227 # 设置K线图标签和图例 

228 if show_text: 

229 ax_kline.set_ylabel('Price', fontsize=9) 

230 if show_legend: 

231 ax_kline.legend(loc='upper left', fontsize=7, framealpha=0.8) 

232 ax_kline.grid(True, alpha=0.2, linestyle='--') 

233 else: 

234 # 无文字版本：简化标签 

235 ax_kline.set_ylabel('') 

236 if show_legend: 

237 # 简化图例，使用更小的字体和透明背景 

238 ax_kline.legend(loc='upper left', fontsize=6, framealpha=0.5, 

239 fancybox=False, edgecolor='none') 

240 ax_kline.grid(True, alpha=0.15, linestyle='--') 

241 # 移除边框 

242 ax_kline.spines['top'].set_visible(False) 

243 ax_kline.spines['right'].set_visible(False) 

244 

245 # 设置x轴刻度 

246 xticks = range(0, len(df), max(1, len(df) // 6)) 

247 xlabels = [df.iloc[i]['date'].strftime('%m-%d') for i in xticks] 

248 ax_kline.set_xticks(xticks) 

249 ax_kline.set_xticklabels([]) # K线图不显示日期，留给成交量图 

250 

251 # 添加图例说明 (仅在显示文字时) 

252 if show_text: 

253 legend_text = "Short|DuoKong|Key" 

254 ax_kline.text(0.02, 0.98, legend_text, transform=ax_kline.transAxes, 

255 fontsize=7, verticalalignment='top', bbox=dict(boxstyle='round', 

256 facecolor='wheat', alpha=0.3)) 

257 

258 # ========== 底部：成交量 ========== 

259 if show_text: 

260 ax_vol = fig.add_subplot(gs[2], sharex=ax_kline) 

261 else: 

262 ax_vol = fig.add_subplot(gs[1], sharex=ax_kline) 

263 

264 # 计算成交量范围 

265 vol_max = df['volume'].max() 

266 ax_vol.set_ylim(0, vol_max * 1.2) 

267 

268 # 绘制成交量柱状图 

269 for i, row in df.iterrows(): 

270 is_up = row['close'] >= row['open'] 

271 color = '#e74c3c' if is_up else '#27ae60' # 与K线同色 

272 

273 # 成交量柱 

274 ax_vol.bar(i, row['volume'], width=0.7, color=color, alpha=0.7, edgecolor=color) 

275 

276 # 设置成交量图标签 

277 if show_text: 

278 ax_vol.set_ylabel('Vol', fontsize=8) 

279 ax_vol.set_xlabel('Date', fontsize=8) 

280 ax_vol.grid(True, alpha=0.15, linestyle='--', axis='y') 

281 

282 # 设置日期标签 (减少标签数量) 

283 xticks = range(0, len(df), max(1, len(df) // 4)) 

284 xlabels = [df.iloc[i]['date'].strftime('%m-%d') for i in xticks] 

285 ax_vol.set_xticks(xticks) 

286 ax_vol.set_xticklabels(xlabels, rotation=30, ha='right', fontsize=7) 

287 else: 

288 # 无文字版本：简化标签和刻度 

289 ax_vol.set_ylabel('') 

290 ax_vol.set_xlabel('') 

291 ax_vol.grid(True, alpha=0.1, linestyle='--', axis='y') 

292 

293 # 最小化日期标签 

294 xticks = range(0, len(df), max(1, len(df) // 3)) 

295 xlabels = [df.iloc[i]['date'].strftime('%m-%d') for i in xticks] 

296 ax_vol.set_xticks(xticks) 

297 ax_vol.set_xticklabels(xlabels, rotation=0, ha='center', fontsize=6) 

298 

299 # 移除边框 

300 ax_vol.spines['top'].set_visible(False) 

301 ax_vol.spines['right'].set_visible(False) 

302 

303 # 保存图片 (使用较低DPI和优化) 

304 timestamp = datetime.now().strftime('%Y%m%d_%H%M%S') 

305 filename = f"{stock_code}_{category}_{timestamp}.png" 

306 filepath = Path(output_dir) / filename 

307 

308 # 无文字版本使用更低DPI进一步压缩 

309 save_dpi = 50 if show_text else 40 

310 plt.savefig(filepath, dpi=save_dpi, bbox_inches='tight', facecolor='white', edgecolor='none', 

311 pil_kwargs={'optimize': True}) 

312 plt.close(fig) 

313 

314 # 使用PIL二次压缩 

315 compress_image(str(filepath), MAX_FILE_SIZE) 

316 

317 return str(filepath)
